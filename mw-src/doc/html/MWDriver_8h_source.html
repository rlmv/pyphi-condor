877  inline void insert_into_set(MWList<int> * list, int tid) {
878  int i, num = list->number();
879  int *t;
880  bool found = false;
881 
882  for (i=0; i<num; i++) {
883  t = (int*)list->Remove();
884  if (*t==tid)
885  found = true;
886  list->Append(t);
887  }
888  if (!found) {
889  t = new int;
890  *t = tid;
891  list->Append(t);
892  }
893  }
895  inline bool in_set(MWList<int> * list, int tid) {
896  int i, num = list->number();
897  int *t;
898 
899  for (i=0; i<num; i++) {
900  t = (int*)list->Remove();
901  list->Append(t);
902  if (*t==tid)
903  return true;
904  }
905  return false;
906  }
